{{- /* relimg — image with a src that resolves to either a page-bundle resource (if the page is a bundle and the file matches), or a static path with relURL applied (so paths work under both production and the GH Pages preview baseURL). Usage in markdown: {{< relimg src="Belson.png" alt="..." >}} ← bundle resource {{< relimg src="images/foo.jpg" alt="..." >}} ← static path */ -}} {{- $src := .Get "src" -}} {{- $alt := .Get "alt" -}} {{- $class := .Get "class" -}} {{- $resource := .Page.Resources.GetMatch $src -}} {{- $url := "" -}} {{- if $resource -}} {{- $url = $resource.RelPermalink -}} {{- else -}} {{- $url = strings.TrimPrefix "/" $src | relURL -}} {{- end -}}
